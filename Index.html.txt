  <script>
    document.getElementById('housekeepingForm')
      .addEventListener('submit', async e => {
        e.preventDefault();

        // Grab values
        const zone  = document.getElementById('zoneSelect').value;
        const name  = document.getElementById('cleanerName').value.trim();
        const tasks = {
          floorSwept: document.getElementById('task1').checked,
          binsEmptied: document.getElementById('task2').checked,
          machinesWiped: document.getElementById('task3').checked,
        };
        const timestamp = new Date();

        // Photo upload (if any)
        let photoUrl = '';
        const file = document.getElementById('photoInput').files[0];
        if (file) {
          const path = `photos/${zone}/${Date.now()}_${file.name}`;
          const snap = await storage.ref(path).put(file);
          photoUrl = await snap.ref.getDownloadURL();
        }

        // Send to Firestore
        await db.collection('logs').add({ zone, name, tasks, timestamp, photoUrl });

        alert(`âœ… Logged for ${zone}! Thanks, ${name}.`);
        e.target.reset();
      });
